CREATE TABLE
    INTEREST_RATE
    (
        LOAN_TYPE VARCHAR(18),
	CREDIT_RATING VARCHAR(5),
	INTEREST_RATE DECIMAL(5, 2),
        PRIMARY KEY (LOAN_TYPE, CREDIT_RATING)
    );

CREATE TABLE
    LOAN
    (
        LOAN_ID BIGINT NOT NULL IDENTITY(10000,1),
	USER_ID BIGINT,
	LOAN_TYPE VARCHAR(18),
	AMOUNT DECIMAL(15, 2),
	EMI DECIMAL(15, 2),
	INSTALLMENTS INT,
    );

CREATE TABLE APPLICATION_USERS
    (	
	USER_ID BIGINT NOT NULL IDENTITY(10000,1),
	FIRST_NAME VARCHAR(18),
	LAST_NAME VARCHAR(18),
	PAN_CARD VARCHAR(10),
	EMAIL VARCHAR(30),
	USERNAME VARCHAR(30) NOT NULL,
	PASSWORD VARCHAR(75),
	AUTHORITIES VARCHAR(20)
    );
